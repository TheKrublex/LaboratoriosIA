def matrixMultiplication(A, B):
    m, n = len(A), len(B)
    p = len(B[0])
    C = [[0 for _ in range(p)] for _ in range(m)]
    for i in range(m):
        for j in range(p):
            C[i][j] = sum([A[i][k] * B[k][j] for k in range(n)])
    return C

def sumBias(M, bias):
  for i in range(len(M)):
    for j in range(len(M[0])):
      M[i][j] += bias[i][0]
  return M

def forward(weights, bias, featuresT):
  weights = weights.tolist()
  bias = bias.tolist()
  featuresT = featuresT.tolist()
  result = matrixMultiplication(weights, featuresT)
  result = sumBias(result, bias) # Agregar la programacin del bias
  result = torch.tensor(result)
  return sigmoidalActivationFunction(result)
